引用
下載並解壓縮後，保持資料夾結構，丟到你的網頁目錄，即可開始使用！

<!-- 洨洨安 CSS -->
<link rel="stylesheet" href="ts.xiaoan.css">
<!-- 洨洨安 JS -->
<script src="ts.xiaoan.js"></script>


外部引用
不放在自家伺服器的方法，這裡有兩個位置可以自由選用

<!------------------ RawGit ------------------>
<!-- 洨洨安 CSS -->
<link rel="stylesheet" href="https://cdn.rawgit.com/gnehs/Tocas-UI-Xiaoan/master/ts.xiaoan.css">
<!-- 洨洨安 JS -->
<script src="https://cdn.rawgit.com/gnehs/Tocas-UI-Xiaoan/master/ts.xiaoan.js"></script>
<!------------------ / RawGit ------------------>
<!------------------ GitHub Page ------------------>
<!-- 洨洨安 CSS -->
<link rel="stylesheet" href="https://xiaoan.tocas-ui.com/ts.xiaoan.css">
<!-- 洨洨安 JS -->
<script src="https://xiaoan.tocas-ui.com/ts.xiaoan.js"></script>
<!------------------ / GitHub Page ------------------>
